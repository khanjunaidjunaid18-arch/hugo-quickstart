        function showSection(id) {
            document.querySelectorAll('.section').forEach(s => s.classList.add('hidden'));
            document.getElementById(`section-${id}`).classList.remove('hidden');
            window.scrollTo(0,0);
            document.getElementById('mobile-menu').classList.add('hidden');
        }

        function toggleMobileMenu() {
            document.getElementById('mobile-menu').classList.toggle('hidden');
        }

        function filterProducts(cat) {
            const grid = document.getElementById('product-grid');
            grid.innerHTML = '';
            const filtered = cat === 'All' ? products : products.filter(p => p.category === cat);
            filtered.forEach(p => {
                grid.innerHTML += `
                    <div class="group cursor-pointer">
                        <div class="aspect-[4/5] bg-zinc-100 overflow-hidden mb-6 relative rounded-2xl">
                            <img src="${p.img}" class="w-full h-full object-cover group-hover:scale-110 transition-transform duration-700" />
                            <div class="absolute inset-0 bg-black/20 opacity-0 group-hover:opacity-100 transition-opacity"></div>
                            <div class="absolute bottom-6 left-6 right-6 translate-y-4 opacity-0 group-hover:translate-y-0 group-hover:opacity-100 transition-all">
                                <button onclick="addToCart(${p.id})" class="w-full bg-white py-4 font-black uppercase text-[10px] tracking-widest hover:bg-orange-600 hover:text-white transition-colors rounded-xl">Add to Cart</button>
                            </div>
                        </div>
                        <div class="flex justify-between items-start px-2">
                            <div>
                                <h4 class="font-black uppercase tracking-tight text-lg">${p.name}</h4>
                                <p class="text-zinc-400 text-[10px] font-bold uppercase tracking-widest">${p.category}</p>
                            </div>
                            <p class="font-black text-orange-600">Rs. ${p.price.toLocaleString()}</p>
                        </div>
                    </div>
                `;
            });
        }

        function addToCart(id) {
            const product = products.find(p => p.id === id);
            const exists = cart.find(item => item.id === id);
            if (exists) exists.quantity++;
            else cart.push({ ...product, quantity: 1 });
            updateCartUI();
            showSection('cart');
        }

        function updateCartUI() {
            const itemsContainer = document.getElementById('cart-items');
            const region = document.getElementById('cust-region').value;
            const shippingCost = region === 'international' ? 2500 : 0;
            
            const totalItems = cart.reduce((acc, i) => acc + i.quantity, 0);
            document.getElementById('cart-count').innerText = totalItems;
            document.getElementById('cart-count').classList.toggle('hidden', totalItems === 0);
            document.getElementById('cart-summary').classList.toggle('hidden', totalItems === 0);

            itemsContainer.innerHTML = '';
            let subtotal = 0;
            
            if (cart.length === 0) {
                itemsContainer.innerHTML = `<div class="py-20 text-center bg-zinc-50 rounded-3xl font-bold uppercase text-zinc-400">Empty</div>`;
            }

            cart.forEach(item => {
                subtotal += item.price * item.quantity;
                itemsContainer.innerHTML += `
                    <div class="flex gap-6 items-center p-4 bg-white border border-zinc-100 rounded-3xl">
                        <img src="${item.img}" class="w-20 h-20 object-cover rounded-xl" />
                        <div class="flex-1">
                            <h4 class="font-black uppercase text-sm">${item.name}</h4>
                            <p class="text-[10px] font-bold text-orange-600 mt-1">Rs. ${item.price.toLocaleString()}</p>
                        </div>
                        <div class="flex items-center gap-3 bg-zinc-50 p-2 rounded-xl">
                            <button onclick="changeQty(${item.id}, -1)" class="w-8 h-8 bg-white border rounded-lg">-</button>
                            <span class="font-black text-xs">${item.quantity}</span>
                            <button onclick="changeQty(${item.id}, 1)" class="w-8 h-8 bg-white border rounded-lg">+</button>
                        </div>
                    </div>
                `;
            });

            const total = subtotal + shippingCost;
            document.getElementById('summary-subtotal').innerText = `Rs. ${subtotal.toLocaleString()}`;
            document.getElementById('summary-shipping').innerText = shippingCost === 0 ? 'FREE' : `Rs. ${shippingCost.toLocaleString()}`;
            document.getElementById('summary-total').innerText = `Rs. ${total.toLocaleString()}`;
        }

        function changeQty(id, delta) {
            const item = cart.find(i => i.id === id);
            item.quantity += delta;
            if (item.quantity <= 0) cart = cart.filter(i => i.id !== id);
            updateCartUI();
        }

        function setPayment(method) {
            activePayment = method;
            document.querySelectorAll('.pay-btn').forEach(b => {
                b.classList.remove('border-orange-600', 'bg-orange-600', 'text-white');
                b.classList.add('border-white/10', 'text-white/50');
            });
            const activeId = method === 'cod' ? 'btn-cod' : 'btn-jazz';
            document.getElementById(activeId).classList.add('border-orange-600', 'bg-orange-600', 'text-white');
            document.getElementById('jazz-info').classList.toggle('hidden', method !== 'jazzcash');
        }

        function completeOrder() {
            const name = document.getElementById('cust-name').value;
            const phone = document.getElementById('cust-phone').value;
            const address = document.getElementById('cust-address').value;

            if (!name || !phone || !address) {
                alert("Please fill in all shipping details first.");
                return;
            }

            document.getElementById('cart-content').classList.add('hidden');
